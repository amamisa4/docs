// ===== utils =====
function uid() {
    // [fix #8] crypto.randomUUID() で衝突リスクを排除
    if (typeof crypto !== 'undefined' && crypto.randomUUID) return crypto.randomUUID();
    return Date.now().toString(36) + Math.random().toString(36).slice(2, 12);
}

function fmtShort(iso) {
    try {
        const d = new Date(iso), now = new Date();
        const hh = String(d.getHours()).padStart(2, '0'), mi = String(d.getMinutes()).padStart(2, '0');
        if (d.toDateString() === now.toDateString()) return hh + ':' + mi;
        return String(d.getMonth() + 1).padStart(2, '0') + '/' + String(d.getDate()).padStart(2, '0') + ' ' + hh + ':' + mi;
    } catch { return ''; }
}

function fmtFull(iso) {
    try {
        const d = new Date(iso);
        return d.getFullYear() + '/' +
            String(d.getMonth() + 1).padStart(2, '0') + '/' +
            String(d.getDate()).padStart(2, '0') + ' ' +
            String(d.getHours()).padStart(2, '0') + ':' +
            String(d.getMinutes()).padStart(2, '0');
    } catch { return ''; }
}

function fmtDateTime(iso) {
    return fmtFull(iso);
}

function esc(s) {
    return (s || '').replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;');
}

function newerIso(a, b) {
    return new Date(a || 0).getTime() >= new Date(b || 0).getTime();
}

function mergeBlocks(existingBlocks = [], incomingBlocks = []) {
    const byId = new Map(existingBlocks.map(b => [b.id, b]));
    for (const incoming of incomingBlocks) {
        const current = byId.get(incoming.id);
        if (!current) {
            byId.set(incoming.id, incoming);
            continue;
        }
        const next = { ...current };
        if (newerIso(incoming.updatedAt, current.updatedAt)) Object.assign(next, incoming);
        else {
            if ('groupId' in incoming) next.groupId = incoming.groupId;
            if ('groupOrder' in incoming) next.groupOrder = incoming.groupOrder;
        }
        byId.set(incoming.id, next);
    }
    return Array.from(byId.values());
}

// エイリアスのキー形式: 大文字アルファベット + 数字 (例: H1, P2, IMG3)
const ALIAS_RE = /\b([A-Z]+[0-9]+)\b/g;

const AliasStore = {
    load() {
        return Store.get('cn_aliases') || [];
    },
    save(list) {
        Store.set('cn_aliases', list);
    }
};

let aliases = AliasStore.load(); // [{ key: 'H1', text: '笑顔にして' }, ...]

function aliasMap() {
    const m = new Map();
    for (const a of aliases) m.set(a.key, a.text);
    return m;
}

// プレーンテキストへの展開（コピー用）
function expandAliases(text) {
    const m = aliasMap();
    return text.replace(ALIAS_RE, (match) => m.has(match) ? m.get(match) : match);
}

// HTML表示用: エイリアス部分を色付きバッジに変換、未定義は赤バッジ
function renderAliasedHTML(text) {
    const m = aliasMap();
    // テキストをエスケープしつつエイリアスだけバッジ化
    // ALIAS_REはglobalなのでlastIndexをリセット
    ALIAS_RE.lastIndex = 0;
    const parts = [];
    let last = 0;
    let match;
    const re = new RegExp(ALIAS_RE.source, 'g');
    while ((match = re.exec(text)) !== null) {
        // マッチ前の通常テキスト
        if (match.index > last) {
            parts.push(esc(text.slice(last, match.index)));
        }
        const key = match[1];
        if (m.has(key)) {
            parts.push('<span class="alias-badge alias-badge--defined" title="' + esc(key) + '">' + esc(m.get(key)) + '</span>');
        } else {
            parts.push('<span class="alias-badge alias-badge--undefined" title="未定義のエイリアス">' + esc(key) + '</span>');
        }
        last = match.index + match[0].length;
    }
    if (last < text.length) parts.push(esc(text.slice(last)));
    return parts.join('');
}

// ===== WebSocket =====
const WS = (() => {
    let ws = null, pingT = null, reconnT = null, watchdogT = null, closed = false, authed = false;
    let lastPongAt = 0;
    const PING_INTERVAL = 20000;   // 20秒ごとにping（DOの30秒タイムアウト前に送る）
    const PONG_TIMEOUT = 10000;   // pong が10秒来なければ死亡と判断
    const pending = new Map();

    function _clearTimers() {
        clearInterval(pingT);
        clearInterval(watchdogT);
        clearTimeout(reconnT);
        pingT = watchdogT = reconnT = null;
    }

    function _forceReconnect() {
        // WSが静かに死んでいる場合の強制再接続
        if (closed) return;
        console.warn('[WS] pong timeout — force reconnect');
        authed = false;
        _clearTimers();
        if (ws) { try { ws.close(); } catch { } ws = null; }
        setStatus('offline');
        reconnT = setTimeout(connect, 500);
    }

    function connect() {
        if (closed) return;
        const { url, key, room } = state.settings;
        if (!url || !room) return;
        try {
            // [fix #1] ?key= をURLから除去。認証は接続後の auth メッセージのみで行う
            ws = new WebSocket(
                url.replace(/^http/, 'ws') + '/ws/' + encodeURIComponent(room)
            );
        } catch { return; }

        ws.onopen = () => {
            // サーバーは接続後に auth メッセージで認証する仕様
            ws.send(JSON.stringify({ type: 'auth', key }));
            authed = false;
            setStatus('syncing');
        };
        ws.onmessage = e => {
            try {
                const m = JSON.parse(e.data);
                if (m.type === 'pong') {
                    lastPongAt = Date.now(); // pong 受信時刻を更新
                    return;
                }
                if (m.type === 'auth_ok') {
                    authed = true;
                    lastPongAt = Date.now();
                    setStatus(pending.size ? 'syncing' : 'online');
                    flushPending();
                    if (pending.size === 0) loadFromServer();
                    _clearTimers();

                    // ping を定期送信
                    pingT = setInterval(() => {
                        if (ws && ws.readyState === 1 && authed) {
                            ws.send(JSON.stringify({ type: 'ping' }));
                        }
                    }, PING_INTERVAL);

                    // watchdog: pong が PONG_TIMEOUT 以上来なければ強制再接続
                    watchdogT = setInterval(() => {
                        if (Date.now() - lastPongAt > PING_INTERVAL + PONG_TIMEOUT) {
                            _forceReconnect();
                        }
                    }, PING_INTERVAL + PONG_TIMEOUT);

                    return;
                }
                if (m.type === 'auth_error') { setStatus('error'); return; }
                if (m.type === 'sync_error') { setStatus('error'); console.warn('sync error', m); return; }
                if (m.type === 'sync') {
                    if (m.clientId === state.clientId && m.clientMsgId) {
                        pending.delete(m.clientMsgId);
                        setStatus(pending.size ? 'syncing' : 'online');
                    }
                    applySync(m, true);
                }
            } catch { }
        };
        ws.onclose = () => {
            authed = false;
            _clearTimers();
            setStatus('offline');
            if (!closed) reconnT = setTimeout(connect, 1500);
        };
        ws.onerror = () => setStatus('error');
    }

    // PCのスリープ復帰・タブ復帰時に即再接続
    document.addEventListener('visibilitychange', () => {
        if (document.visibilityState === 'visible' && !closed) {
            const alive = ws && ws.readyState === 1 && authed
                && Date.now() - lastPongAt < PING_INTERVAL + PONG_TIMEOUT;
            if (!alive) {
                console.log('[WS] visibility restored — reconnecting');
                _forceReconnect();
            }
        }
    });

    function prepare(m) {
        if (m.type !== 'sync') return m;
        if (!m.clientId) m.clientId = state.clientId;
        if (!m.clientMsgId) m.clientMsgId = uid();
        if (!m.clientSeq) m.clientSeq = ++state.syncSeq;
        pending.set(m.clientMsgId, m);
        setStatus('syncing');
        return m;
    }

    function transmit(m) {
        try {
            if (ws && ws.readyState === 1 && authed) ws.send(JSON.stringify(m));
        } catch {
            setStatus('error');
        }
    }

    function flushPending() {
        if (!authed) return;
        pending.forEach(m => transmit(m));
    }

    function send(m) {
        m = prepare(m);
        if (m.type !== 'sync') { transmit(m); return; }
        if (authed) transmit(m);
    }
    function disconnect() { closed = true; authed = false; _clearTimers(); if (ws) ws.close(); }
    function reset() { closed = false; disconnect(); closed = false; setTimeout(connect, 300); }

    function ensureConnected() {
        if (closed) return;
        const alive = ws && ws.readyState === 1 && authed
            && Date.now() - lastPongAt < PING_INTERVAL + PONG_TIMEOUT;
        if (!alive) _forceReconnect();
    }

    return { connect, send, disconnect, reset, ensureConnected };
})();

function setStatus(s) {
    const d = document.getElementById('statusDot');
    d.className = 'sidebar-status ' + (s === 'online' ? 'online' : s === 'error' ? 'error' : s === 'syncing' ? 'syncing' : '');
    d.title = s === 'online' ? '接続中' : s === 'syncing' ? '同期中' : s === 'error' ? 'エラー' : '未接続';
}

function applySync(msg, fromRemote) {
    const { action, payload } = msg;
    if (action === 'note_upsert') {
        const idx = state.notes.findIndex(n => n.id === payload.id);
        if (idx >= 0) {
            const existing = state.notes[idx];
            if (!newerIso(payload.updatedAt, existing.updatedAt)) {
                if (fromRemote) {
                    const merged = { ...existing };
                    if (payload.blocks) merged.blocks = mergeBlocks(existing.blocks, payload.blocks);
                    state.notes[idx] = merged;
                }
            } else {
                const merged = { ...payload };
                if (payload.blocks && existing.blocks) merged.blocks = mergeBlocks(existing.blocks, payload.blocks);
                state.notes[idx] = merged;
            }
        } else {
            state.notes.push(payload);
        }
        sortNotes(); renderNoteList();
        if (state.currentNoteId === payload.id) {
            renderHeader();
            if (!getActiveEditingBlockId()) renderBlocks();
        }
    } else if (action === 'note_delete') {
        state.notes = state.notes.filter(n => n.id !== payload.id);
        if (state.currentNoteId === payload.id) {
            state.currentNoteId = state.notes[0]?.id || null;
            renderAll();
        } else { renderNoteList(); }
    } else if (action === 'block_upsert') {
        const note = state.notes.find(n => n.id === payload.noteId);
        if (!note) return;
        const idx = note.blocks.findIndex(b => b.id === payload.id);
        if (idx >= 0) {
            const cur = note.blocks[idx];
            if (newerIso(payload.updatedAt, cur.updatedAt)) note.blocks[idx] = { ...cur, ...payload };
            else {
                if ('groupId' in payload) note.blocks[idx] = { ...cur, groupId: payload.groupId, groupOrder: payload.groupOrder };
            }
        } else { note.blocks.push(payload); }
        if (newerIso(payload.noteUpdatedAt, note.updatedAt)) note.updatedAt = payload.noteUpdatedAt;
        if (state.currentNoteId === payload.noteId) {
            renderHeader();
            const editingId = getActiveEditingBlockId();
            syncActiveTextarea(note.blocks.find(b => b.id === payload.id), editingId === payload.id);
            if (editingId !== payload.id) renderBlocks();
        }
        renderNoteList();
    } else if (action === 'block_delete') {
        const note = state.notes.find(n => n.id === payload.noteId);
        if (!note) return;
        note.blocks = note.blocks.filter(b => b.id !== payload.id);
        if (newerIso(payload.noteUpdatedAt, note.updatedAt)) note.updatedAt = payload.noteUpdatedAt;
        if (state.currentNoteId === payload.noteId) { renderHeader(); renderBlocks(); }
        renderNoteList();
    }
}

function dispatch(action, payload) {
    const now = new Date().toISOString();
    if (action === 'note_upsert') {
        payload.updatedAt = payload.updatedAt || now;
        applySync({ action, payload }, false);
        WS.send({ type: 'sync', action, payload });
    } else if (action === 'note_delete') {
        applySync({ action, payload }, false);
        WS.send({ type: 'sync', action, payload });
    } else if (action === 'block_upsert') {
        payload.createdAt = payload.createdAt || now;
        payload.updatedAt = now;
        payload.noteUpdatedAt = now;
        applySync({ action, payload }, false);
        WS.send({ type: 'sync', action, payload });
    } else if (action === 'block_delete') {
        payload.noteUpdatedAt = now;
        applySync({ action, payload }, false);
        WS.send({ type: 'sync', action, payload });
    }
}

async function loadFromServer() {
    const { url, key, room } = state.settings; if (!url || !room) return;
    try {
        const r = await fetch(url + '/notes/' + encodeURIComponent(room), { headers: { 'x-api-key': key } });
        if (!r.ok) return;
        const data = await r.json();
        for (const n of (data.notes || [])) applySync({ action: 'note_upsert', payload: n }, true);
        sortNotes(); renderAll();
    } catch { }
}

function sortNotes() {
    state.notes.sort((a, b) => {
        if (a.pinned !== b.pinned) return a.pinned ? -1 : 1;
        return new Date(b.updatedAt || 0) - new Date(a.updatedAt || 0);
    });
}

function currentNote() { return state.notes.find(n => n.id === state.currentNoteId) || null; }

function getItemOrder(note) {
    if (note.itemOrder && note.itemOrder.length > 0) return note.itemOrder;
    const pinned = (note.blocks || []).filter(b => b.pinned && !b.groupId).map(b => ({ type: 'block', id: b.id }));
    const groups = (note.groups || []).map(g => ({ type: 'group', id: g.id }));
    const ungrouped = (note.blocks || []).filter(b => !b.pinned && !b.groupId).map(b => ({ type: 'block', id: b.id }));
    return [...pinned, ...groups, ...ungrouped];
}

function getGroupBlocks(note, groupId) {
    return (note.blocks || [])
        .filter(b => b.groupId === groupId)
        .sort((a, b) => (a.groupOrder ?? 0) - (b.groupOrder ?? 0));
}

function renderNoteList() {
    const q = (document.getElementById('searchInput').value || '').toLowerCase();
    const el = document.getElementById('noteList');
    el.innerHTML = '';
    for (const n of state.notes) {
        const title = n.title || '無題のメモ';
        if (q && !title.toLowerCase().includes(q) && !(n.blocks || []).some(b => b.text.toLowerCase().includes(q))) continue;
        const d = document.createElement('div');
        d.className = 'note-item' + (n.id === state.currentNoteId ? ' active' : '') + (n.pinned ? ' pinned' : '');
        d.innerHTML = '<div class="note-item-title">' + (n.pinned ? '<svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" width="11" height="11"><path d="M12 17v5M9 3h6l1 7H8L9 3zM8 10l-2 7h12l-2-7"/></svg>' : '') + esc(title) + '</div>' +
            '<div class="note-item-meta">' + fmtShort(n.updatedAt) + ' · ' + (n.blocks || []).length + 'ブロック</div>';
        d.addEventListener('click', () => { state.currentNoteId = n.id; renderAll(); App.hideSidebar(); });
        el.appendChild(d);
    }
}

function renderHeader() {
    const n = currentNote();
    const titleInput = document.getElementById('noteTitleInput');
    const updated = document.getElementById('noteUpdated');
    const addBar = document.getElementById('addBar');
    const pinBtn = document.getElementById('notePinBtn');
    if (!n) {
        titleInput.value = ''; titleInput.disabled = true;
        updated.textContent = '';
        addBar.style.display = 'none';
        pinBtn.classList.remove('active');
        return;
    }
    titleInput.disabled = false;
    if (document.activeElement !== titleInput) titleInput.value = n.title || '';
    updated.textContent = fmtDateTime(n.updatedAt);
    addBar.style.display = 'flex';
    pinBtn.classList.toggle('active', !!n.pinned);
}

function renderBlocks() {
    const note = currentNote();
    const area = document.getElementById('blocksArea');
    if (!note) {
        area.innerHTML = '<div class="empty-state"><svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="1.5"><path d="M14 2H6a2 2 0 0 0-2 2v16a2 2 0 0 0 2 2h12a2 2 0 0 0 2-2V8z"/><polyline points="14 2 14 8 20 8"/></svg>メモを選択するか、新しく作成してください</div>';
        return;
    }
    if (state.sortMode) { initSortDrag(area, note); return; }
    area.innerHTML = '';
    const order = getItemOrder(note);
    for (const item of order) {
        if (item.type === 'block') {
            const b = (note.blocks || []).find(bl => bl.id === item.id);
            if (b) area.appendChild(makeBlockEl(b, note));
        } else if (item.type === 'group') {
            const g = (note.groups || []).find(gr => gr.id === item.id);
            if (g) {
                const gblocks = getGroupBlocks(note, g.id);
                area.appendChild(makeGroupEl(g, gblocks, note));
            }
        }
    }
    if (!state.sortMode) initDragDrop(area, note);
}

function makeGroupEl(g, blocks, note) {

    el.className = 'block-group';
    el.dataset.id = g.id;
    el.dataset.type = 'group';

    const header = document.createElement('div');
    header.className = 'group-header';

    const titleEl = document.createElement('div');
    titleEl.className = 'group-title';
    titleEl.textContent = g.title || 'グループ';
    titleEl.contentEditable = 'false';

    function saveTitle() {
        const newTitle = titleEl.textContent.trim() || 'グループ';
        titleEl.contentEditable = 'false';
        const n = currentNote(); if (!n) return;
        const updated = { ...n, groups: n.groups.map(gr => gr.id === g.id ? { ...gr, title: newTitle } : gr), updatedAt: new Date().toISOString() };
        state.notes = state.notes.map(no => no.id === n.id ? updated : no);
        persist(); renderHeader(); renderNoteList(); WS.send({ type: 'sync', action: 'note_upsert', payload: updated });
    }

    titleEl.addEventListener('dblclick', () => {
        titleEl.contentEditable = 'true'; titleEl.focus();
        const range = document.createRange(); range.selectNodeContents(titleEl);
        window.getSelection().removeAllRanges(); window.getSelection().addRange(range);
    });

    titleEl.addEventListener('keydown', e => { if (e.key === 'Enter') { e.preventDefault(); titleEl.blur(); } });

    const delBtn = document.createElement('button');
    delBtn.className = 'group-del-btn';
    delBtn.title = 'グループを解除';
    delBtn.innerHTML = '<svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><path d="M18 6L6 18M6 6l12 12"/></svg>';
    delBtn.addEventListener('click', () => {
        const n = currentNote(); if (!n) return;
        const newBlocks = n.blocks.map(b => b.groupId === g.id ? { ...b, groupId: null, groupOrder: null } : b);
        const newGroups = (n.groups || []).filter(gr => gr.id !== g.id);

            .filter(it => !(it.type === 'group' && it.id === g.id))
            .concat(blocks.map(b => ({ type: 'block', id: b.id })));
        const updated = { ...n, blocks: newBlocks, groups: newGroups, itemOrder: newOrder, updatedAt: new Date().toISOString() };
        state.notes = state.notes.map(no => no.id === n.id ? updated : no);
        persist(); renderHeader(); renderBlocks(); renderNoteList(); WS.send({ type: 'sync', action: 'note_upsert', payload: updated });
    });

    header.appendChild(titleEl);
    header.appendChild(delBtn);
    el.appendChild(header);

    const inner = document.createElement('div');
    inner.className = 'group-inner';
    inner.dataset.groupId = g.id;

    for (const b of blocks) inner.appendChild(makeBlockEl(b, note));

    // グループ内ブロック追加
    const addRow = document.createElement('div');
    addRow.className = 'group-add-row';
    const ta = document.createElement('textarea');
    ta.className = 'group-add-ta'; ta.rows = 1; ta.placeholder = 'グループに追加...';
    ta.addEventListener('input', function () {
        this.style.height = '32px'; this.style.height = Math.min(this.scrollHeight, 100) + 'px';
    });

    function doAdd() {
        const text = ta.value.trim(); if (!text) return;
        const n = currentNote(); if (!n) return;
        const now = new Date().toISOString();
        const maxOrder = Math.max(0, ...blocks.map(b => b.groupOrder ?? 0));
        const block = { id: uid(), noteId: n.id, text, pinned: false, createdAt: now, groupId: g.id, groupOrder: maxOrder + 1 };
        const { blocks: _, ...noteWithoutBlocks } = { ...n, updatedAt: now };
        dispatch('block_upsert', block);
        dispatch('note_upsert', noteWithoutBlocks);
        ta.value = ''; ta.style.height = '32px';
    }
    ta.addEventListener('keydown', e => { if (e.key === 'Enter' && !e.shiftKey) { e.preventDefault(); doAdd(); } });
    const addBtn = document.createElement('button');
    addBtn.className = 'group-add-btn';
    addBtn.innerHTML = '<svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><path d="M12 5v14M5 12h14"/></svg>';
    addBtn.addEventListener('click', doAdd);
    addRow.appendChild(ta); addRow.appendChild(addBtn);

    inner.appendChild(addRow);
    el.appendChild(inner);
    if (!state.sortMode) initDragDrop(inner, note);
    return el;
}

function makeBlockEl(b, note) {
    const card = document.createElement('div');
    card.className = 'block-card' + (b.pinned ? ' pinned' : '');
    card.dataset.id = b.id;
    card.dataset.type = 'block';
    card.draggable = false;

    const groups = note.groups || [];
    const pinCls = 'block-act' + (b.pinned ? ' pin-on' : '');
    const pinLbl = b.pinned ? 'ピン留めを外す' : 'ピン留め';
    const grpCls = 'block-act' + (b.groupId ? ' grp-on' : '');
    const pickerItems = groups.map(g =>
        '<div class="grp-picker-item' + (b.groupId === g.id ? ' cur' : '') + '" data-gid="' + g.id + '">' + esc(g.title || 'グループ') + '</div>'

// ===== FLIP animation helper =====
function flipAnimate(area, fn) {
    // First: 現在位置を記録
    const rects = new Map();
    area.querySelectorAll(':scope > .block-card, :scope > .block-group').forEach(el => {
        rects.set(el.dataset.id, el.getBoundingClientRect());
    });

    fn(); // DOM更新（renderBlocks を含む）

    // Last: 新位置を取得し、差分でアニメーション
    area.querySelectorAll(':scope > .block-card, :scope > .block-group').forEach(el => {
        const old = rects.get(el.dataset.id);
        if (!old) return;
        const now = el.getBoundingClientRect();
        const dy = old.top - now.top;
        if (Math.abs(dy) < 1) return;
        el.style.transform = `translateY(${dy}px)`;
        el.style.transition = 'none';
        requestAnimationFrame(() => {
            el.style.transition = 'transform .22s ease';
            el.style.transform = '';
            el.addEventListener('transitionend', () => { el.style.transition = ''; }, { once: true });
        });
    });
}

function getOrCreateIndicator() {
    let ind = document.getElementById('drop-indicator');
    if (!ind) {
        ind = document.createElement('div');
        ind.id = 'drop-indicator';
        ind.className = 'drop-indicator';
        document.body.appendChild(ind);
    }
    return ind;
}

function showIndicator(targetEl, before) {
    const ind = getOrCreateIndicator();
    const rect = targetEl.getBoundingClientRect();
    const scrollY = window.scrollY || document.documentElement.scrollTop;
    ind.style.top = (before ? rect.top : rect.bottom) + scrollY - 1.5 + 'px';
    ind.style.left = rect.left + 'px';
    ind.style.width = rect.width + 'px';
    ind.style.display = 'block';
}

function hideIndicator() {
    const ind = document.getElementById('drop-indicator');
    if (ind) ind.style.display = 'none';
}

function initDragDrop(area, note) {
    let dragId = null, dragType = null, overEl = null, overBefore = false;

    area.querySelectorAll(':scope > .block-card, :scope > .block-group').forEach(el => {
        el.setAttribute('draggable', 'false');
    });

    function getDragItems() {
        return Array.from(area.querySelectorAll(':scope > .block-card, :scope > .block-group'));
    }

    function onDragStart(e, el) {
        dragId = el.dataset.id; dragType = el.dataset.type;
        el.classList.add('dragging');
        e.dataTransfer.effectAllowed = 'move';
    }

    function onDragOver(e, el) {
        if (!dragId || el.dataset.id === dragId) return;
        e.preventDefault();
        const rect = el.getBoundingClientRect();
        overBefore = e.clientY < rect.top + rect.height / 2;
        overEl = el;
        showIndicator(el, overBefore);
    }

    function onDrop(e) {
        e.preventDefault();
        hideIndicator();
        if (!dragId || !overEl || overEl.dataset.id === dragId) { cleanup(); return; }
        const toId = overEl.dataset.id;
        const toType = overEl.dataset.type;
        flipAnimate(area, () => applyTopLevelReorder(note, dragId, dragType, toId, toType, overBefore));
        cleanup();
    }

    function cleanup() {
        area.querySelectorAll('.dragging').forEach(el => el.classList.remove('dragging'));
        hideIndicator(); dragId = dragType = overEl = null;
    }

    getDragItems().forEach(el => {
        const handle = el.querySelector('.sort-handle');
        if (handle) {
            handle.addEventListener('mousedown', () => el.setAttribute('draggable', 'true'));
            handle.addEventListener('mouseup', () => el.setAttribute('draggable', 'false'));
        }
        el.addEventListener('dragstart', e => onDragStart(e, el));
        el.addEventListener('dragover', e => onDragOver(e, el));
        el.addEventListener('dragleave', () => { hideIndicator(); overEl = null; });
        el.addEventListener('drop', onDrop);
        el.addEventListener('dragend', cleanup);
    });
}

function applyGroupBlockReorder(note, groupId, fromId, toId, before = false) {
    const blocks = getGroupBlocks(note, groupId);
    const fromIdx = blocks.findIndex(b => b.id === fromId);
    const toIdx = blocks.findIndex(b => b.id === toId);
    if (fromIdx < 0 || toIdx < 0) return;
    const moved = blocks.splice(fromIdx, 1)[0];
    const insertAt = before ? toIdx : toIdx + 1;
    blocks.splice(insertAt > fromIdx ? insertAt - 1 : insertAt, 0, moved);
    const newBlocks = note.blocks.map(b => {
        const idx = blocks.findIndex(bl => bl.id === b.id);
        return idx >= 0 ? { ...b, groupOrder: idx } : b;
    });
    const updated = { ...note, blocks: newBlocks, updatedAt: new Date().toISOString() };
    state.notes = state.notes.map(n => n.id === note.id ? updated : n);
    persist(); renderHeader(); renderBlocks(); renderNoteList(); WS.send({ type: 'sync', action: 'note_upsert', payload: updated });
}

function initSortDrag(area, note) {
    area.innerHTML = '';
    const order = getItemOrder(note);
    for (const item of order) {
        if (item.type === 'block') {
            const b = (note.blocks || []).find(bl => bl.id === item.id);
            if (b) area.appendChild(makeBlockEl(b, note));
        } else if (item.type === 'group') {
            const g = (note.groups || []).find(gr => gr.id === item.id);
            if (g) {
                const gblocks = getGroupBlocks(note, g.id);
                area.appendChild(makeGroupEl(g, gblocks, note));
            }
        }
    }

    let dragId = null, dragType = null;

    function getTopLevelEl(el) {
        let cur = el;
        while (cur && cur.parentElement !== area) cur = cur.parentElement;
        return cur;
    }

    function startDrag(clientX, clientY, el, type) {
        dragId = el.dataset.id; dragType = type;
        el.classList.add('sort-dragging');
    }

    function moveDrag(clientX, clientY) {
        if (!dragId) return;
        const items = Array.from(area.querySelectorAll(':scope > .block-card, :scope > .block-group'));
        let target = null, before = false;
        for (const item of items) {
            if (item.dataset.id === dragId) continue;
            const rect = item.getBoundingClientRect();
            if (clientY < rect.bottom) {
                target = item;
                before = clientY < rect.top + rect.height / 2;
                break;
            }
        }
        if (target) showIndicator(target, before);
        else hideIndicator();
    }

    function endDrag() {
        if (!dragId) return;
        const ind = document.getElementById('drop-indicator');
        hideIndicator();
        const items = Array.from(area.querySelectorAll(':scope > .block-card, :scope > .block-group'));
        let toId = null, before = false;
        let minDist = Infinity;
        for (const item of items) {
            if (item.dataset.id === dragId) continue;
            const rect = item.getBoundingClientRect();
        }
        // Find by indicator position
        if (ind && ind.style.display !== 'none') {
            const indTop = parseFloat(ind.style.top);
            for (const item of items) {
                if (item.dataset.id === dragId) continue;
                const rect = item.getBoundingClientRect();
                const scrollY = window.scrollY || document.documentElement.scrollTop;
                const topDiff = Math.abs(rect.top + scrollY - 1.5 - indTop);
                const botDiff = Math.abs(rect.bottom + scrollY - 1.5 - indTop);
                if (topDiff < 5) { toId = item.dataset.id; before = true; break; }
                if (botDiff < 5) { toId = item.dataset.id; before = false; break; }
            }
        }
        area.querySelectorAll('.sort-dragging').forEach(el => el.classList.remove('sort-dragging'));
        dragId = dragType = null;
        if (toId) {
            const toType = area.querySelector('[data-id="' + toId + '"]')?.dataset.type;
            flipAnimate(area, () => applyTopLevelReorder(note, dragId, dragType, toId, toType, before));
        }
    }

    function onTouchStart(e, el, type) {
        e.preventDefault();
        startDrag(e.touches[0].clientX, e.touches[0].clientY, el, type);
    }
    function onTouchMove(e) {
        e.preventDefault();
        moveDrag(e.touches[0].clientX, e.touches[0].clientY);
    }
    function onTouchEnd() { endDrag(); }

    function onMouseStart(e, el, type) {
        if (e.button !== 0) return;
        startDrag(e.clientX, e.clientY, el, type);
        document.addEventListener('mousemove', onMouseMove);
        document.addEventListener('mouseup', onMouseEnd, { once: true });
    }
    function onMouseMove(e) { moveDrag(e.clientX, e.clientY); }
    function onMouseEnd() {
        document.removeEventListener('mousemove', onMouseMove);
        endDrag();
    }

    area.querySelectorAll(':scope > .block-card, :scope > .block-group').forEach(el => {
        const type = el.dataset.type;
        const handle = el.querySelector('.sort-handle');
        if (handle) {
            handle.addEventListener('touchstart', e => onTouchStart(e, el, type), { passive: false });
            handle.addEventListener('mousedown', e => onMouseStart(e, el, type));
        }
        el.addEventListener('dragover', e => { e.preventDefault(); moveDrag(e.clientX, e.clientY); });
        el.addEventListener('drop', e => { e.preventDefault(); endDrag(); });
    });

    area._sortCleanup = () => {
        document.removeEventListener('mousemove', onMouseMove);
        hideIndicator();
    };
}

function applyTopLevelReorder(note, fromId, fromType, toId, toType, before = false) {
    const order = getItemOrder(note);
    const fromIdx = order.findIndex(it => it.id === fromId);
    const toIdx = order.findIndex(it => it.id === toId);
    if (fromIdx < 0 || toIdx < 0) return;
    const moved = order.splice(fromIdx, 1)[0];
    const insertAt = before ? toIdx : toIdx + 1;
    order.splice(insertAt > fromIdx ? insertAt - 1 : insertAt, 0, moved);
    const updated = { ...note, itemOrder: order, updatedAt: new Date().toISOString() };
    state.notes = state.notes.map(n => n.id === note.id ? updated : n);
    persist(); renderHeader(); renderBlocks(); renderNoteList(); WS.send({ type: 'sync', action: 'note_upsert', payload: updated });
}

function syncActiveTextarea(block, shouldUpdateText) {
    if (!shouldUpdateText) return;
    const card = document.querySelector('.block-card[data-id="' + block.id + '"]');
    const ta = card?.querySelector('.block-edit-ta');
    if (!ta || document.activeElement !== ta || ta.value === block.text) return;
    const start = ta.selectionStart;
    const end = ta.selectionEnd;
    const delta = (block.text || '').length - ta.value.length;
    ta.value = block.text || '';
    const nextStart = Math.max(0, Math.min(ta.value.length, start + delta));
    const nextEnd = Math.max(nextStart, Math.min(ta.value.length, end + delta));
    ta.setSelectionRange(nextStart, nextEnd);
    ta.style.height = 'auto';
    ta.style.height = ta.scrollHeight + 'px';
}

                const text = await file.text();
                const data = JSON.parse(text);
                const now = new Date().toISOString();

                const incoming = Array.isArray(data.notes) ? data.notes : [data];
                if (!incoming.length || !incoming[0].id) {
                    showToast('無効なファイル形式です'); return;
                }

                let added = 0;
                for (const note of incoming) {
                    const noteId = uid();
                    const idMap = {};

                    const newBlocks = (note.blocks || []).map(b => {
                        const newId = uid();
                        idMap[b.id] = newId;
                        return { ...b, id: newId, noteId, groupId: b.groupId || null };
                    });

                    const newGroups = (note.groups || []).map(g => {
                        const newId = uid();
                        idMap[g.id] = newId;
                        return { ...g, id: newId };
                    });

                    const remappedBlocks = newBlocks.map(b => ({
                        ...b,
                        groupId: b.groupId ? (idMap[b.groupId] || null) : null,
                    }));

                    const newItemOrder = (note.itemOrder || [])
                        .map(item => ({ ...item, id: idMap[item.id] }))
                        .filter(item => item.id);

                    const importedNote = {
                        id: noteId,
                        title: (note.title || '無題のメモ') + ' (インポート)',
                        pinned: false,
                        updatedAt: now,
                        blocks: remappedBlocks,
                        groups: newGroups,
                        itemOrder: newItemOrder,
                    };

                    dispatch('note_upsert', importedNote);
                    for (const b of remappedBlocks) {
                        dispatch('block_upsert', { ...b, noteId });
                    }
                    added++;
                }
                showToast(`インポート完了: ${added}件追加`);
                sortNotes(); renderAll();
            } catch (e) {
                console.warn('import error', e);
                showToast('インポートに失敗しました');
            }
        });
        input.click();
    },

    // ===== Alias Modal =====
    openAliasModal() {
        aliases = AliasStore.load();
        renderAliasTable();
        document.getElementById('aliasModal').classList.remove('hidden');
    },
    closeAliasModal() {
        document.getElementById('aliasModal').classList.add('hidden');
    },
    addAlias() {
        const keyEl = document.getElementById('aliasNewKey');
        const textEl = document.getElementById('aliasNewText');
        const key = keyEl.value.trim().toUpperCase();
        const text = textEl.value.trim();
        if (!key || !text) { showToast('キーとテキストを入力してください'); return; }
        if (!/^[A-Z]+[0-9]+$/.test(key)) { showToast('キーは大文字アルファベット+数字で入力してください (例: H1, P2)'); return; }
        const existing = aliases.findIndex(a => a.key === key);
        if (existing >= 0) {
            aliases[existing].text = text;
        } else {
            aliases.push({ key, text });
        }
        aliases.sort((a, b) => a.key.localeCompare(b.key, undefined, { numeric: true }));
        AliasStore.save(aliases);
        keyEl.value = ''; textEl.value = '';
        renderAliasTable();
        renderBlocks(); // 展開表示を即時更新
        showToast(existing >= 0 ? key + ' を更新しました' : key + ' を追加しました');
    },
    deleteAlias(key) {
        aliases = aliases.filter(a => a.key !== key);
        AliasStore.save(aliases);
        renderAliasTable();
        renderBlocks();
    }
};

function renderAliasTable() {
    const tbody = document.getElementById('aliasTbody');
    tbody.innerHTML = '';
    if (aliases.length === 0) {
        tbody.innerHTML = '<tr><td colspan="3" class="alias-empty">エイリアスが登録されていません</td></tr>';
        return;
    }
    for (const a of aliases) {
        const tr = document.createElement('tr');
        tr.innerHTML =
            '<td><span class="alias-key-cell">' + esc(a.key) + '</span></td>' +
            '<td class="alias-text-cell">' + esc(a.text) + '</td>' +
            '<td><button class="alias-del-btn" data-key="' + esc(a.key) + '" title="削除">' +
            '<svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" width="14" height="14"><polyline points="3 6 5 6 21 6"/><path d="M19 6l-1 14H6L5 6"/><path d="M10 11v6M14 11v6"/><path d="M9 6V4h6v2"/></svg>' +
            '</button></td>';
        tr.querySelector('.alias-del-btn').addEventListener('click', () => App.deleteAlias(a.key));
        tbody.appendChild(tr);
    }
}

/**
 * ClipNote Server — Cloudflare Workers
 *
 * 環境変数:
 *   API_KEY : 認証キー (wrangler secret put API_KEY)
 *
 * エンドポイント:
 *   WS   /ws/:roomId            リアルタイム同期
 *   GET  /notes/:roomId         メモ一覧 + ブロック全取得
 *   GET  /ping                  ヘルスチェック
 */

export { NoteRoom } from './room.js';

const INIT_SQL = `
CREATE TABLE IF NOT EXISTS notes (
  id         TEXT PRIMARY KEY,
  room_id    TEXT NOT NULL,
  title      TEXT NOT NULL DEFAULT '',
  pinned     INTEGER NOT NULL DEFAULT 0,
  updated_at TEXT NOT NULL,
  item_order TEXT
);
CREATE TABLE IF NOT EXISTS groups (
  id         TEXT PRIMARY KEY,
  note_id    TEXT NOT NULL,
  room_id    TEXT NOT NULL,
  title      TEXT NOT NULL DEFAULT 'グループ',
  created_at TEXT NOT NULL
);
CREATE TABLE IF NOT EXISTS blocks (
  id          TEXT PRIMARY KEY,
  note_id     TEXT NOT NULL,
  room_id     TEXT NOT NULL,
  text        TEXT NOT NULL DEFAULT '',
  pinned      INTEGER NOT NULL DEFAULT 0,
  created_at  TEXT NOT NULL,
  updated_at  TEXT NOT NULL,
  group_id    TEXT,
  group_order INTEGER
);
CREATE INDEX IF NOT EXISTS idx_notes_room   ON notes(room_id, updated_at);
CREATE INDEX IF NOT EXISTS idx_groups_note  ON groups(note_id);
CREATE INDEX IF NOT EXISTS idx_blocks_note  ON blocks(note_id, created_at);
`;

const _dbInitMap = new WeakMap();
async function ensureDb(env) {
  if (!env.DB) return;
  if (!_dbInitMap.has(env)) {
    const promise = env.DB.exec(INIT_SQL).catch(e => {
      _dbInitMap.delete(env);
      throw e;
    });
    _dbInitMap.set(env, promise);
  }
  await _dbInitMap.get(env);
}

export default {
  async fetch(request, env) {
    const url = new URL(request.url);
    const parts = url.pathname.split('/').filter(Boolean);

    if (request.method === 'OPTIONS') return cors(new Response(null, { status: 204 }), env);
    if (url.pathname === '/ping') return cors(new Response('pong'), env);

    const isWsUpgrade = parts[0] === 'ws' && parts[1] &&
      request.headers.get('Upgrade') === 'websocket';

    if (!isWsUpgrade) {
      const authErr = checkAuth(request, env);
      if (authErr) return cors(authErr, env);
    }

    // [fix #12] DB初期化エラーを握り潰さず、失敗時はログに残す
    try {
      await ensureDb(env);
    } catch (e) {
      console.error('DB initialization failed:', e);
      // DB初期化失敗でもWS接続は継続させる（DO側でも初期化を試みる）
    }

    // WebSocket → Durable Object へ
    if (isWsUpgrade) {
      const id = env.NOTE_ROOM.idFromName(parts[1]);
      const headers = new Headers(request.headers);
      headers.set('X-Api-Key-Expected', env.API_KEY || '');
      const newReq = new Request(request.url, { headers, cf: request.cf });
      const doResp = await env.NOTE_ROOM.get(id).fetch(newReq);
      if (doResp.status === 101) return doResp;
      return cors(doResp, env);
    }

    // REST: メモ + ブロック + グループ一括取得
    if (request.method === 'GET' && parts[0] === 'notes' && parts[1]) {
      return cors(await handleLoad(env, parts[1]), env);
    }

    return cors(new Response('not found', { status: 404 }), env);
  },
};

// ---------- REST handlers ----------

async function handleLoad(env, roomId) {
  if (!env.DB) return jsonResp({ error: 'DB not configured' }, 500);

  const [notesRes, blocksRes, groupsRes] = await Promise.all([
    env.DB.prepare(
      `SELECT * FROM notes WHERE room_id=? ORDER BY pinned DESC, updated_at DESC`
    ).bind(roomId).all(),
    env.DB.prepare(
      `SELECT * FROM blocks WHERE room_id=? ORDER BY note_id, created_at DESC`
    ).bind(roomId).all(),
    env.DB.prepare(
      `SELECT * FROM groups WHERE room_id=?`
    ).bind(roomId).all(),
  ]);

  const blocksByNote = {};
  for (const b of (blocksRes.results || [])) {
    if (!blocksByNote[b.note_id]) blocksByNote[b.note_id] = [];
    blocksByNote[b.note_id].push({
      id: b.id,
      noteId: b.note_id,
      text: b.text,
      pinned: b.pinned === 1,
      createdAt: b.created_at,
      updatedAt: b.updated_at,
      groupId: b.group_id || null,
      groupOrder: b.group_order ?? null,
    });
  }

  const groupsByNote = {};
  for (const g of (groupsRes.results || [])) {
    if (!groupsByNote[g.note_id]) groupsByNote[g.note_id] = [];
    groupsByNote[g.note_id].push({
      id: g.id,
      title: g.title,
    });
  }

  const notes = (notesRes.results || []).map(n => {
    // [fix #11] item_order の JSON パースを安全に行う
    let itemOrder = [];
    if (n.item_order) {
      try {
        itemOrder = JSON.parse(n.item_order);
      } catch (e) {
        console.warn('Failed to parse item_order for note', n.id, e);
      }
    }
    return {
      id: n.id,
      title: n.title,
      pinned: n.pinned === 1,
      updatedAt: n.updated_at,
      itemOrder,
      blocks: blocksByNote[n.id] || [],
      groups: groupsByNote[n.id] || [],
    };
  });

  return jsonResp({ notes });
}

// ---------- Auth ----------

function checkAuth(request, env) {
  // [fix #3] API_KEY 未設定の場合は設定ミスとして拒否する
  if (!env.API_KEY) {
    console.error('API_KEY is not configured. All requests are rejected.');
    return jsonResp({ error: 'server misconfigured' }, 500);
  }
  const key = request.headers.get('X-Api-Key');
  if (key !== env.API_KEY) return jsonResp({ error: 'unauthorized' }, 401);
  return null;
}

// ---------- Helpers ----------

function jsonResp(data, status = 200) {
  return new Response(JSON.stringify(data), {
    status,
    headers: { 'Content-Type': 'application/json' },
  });
}

// [fix #5] CORS を環境変数 ALLOWED_ORIGIN で制限可能にする
// ALLOWED_ORIGIN 未設定時は開発用途として '*' にフォールバック（本番では必ず設定すること）
function cors(res, env) {
  const origin = (env && env.ALLOWED_ORIGIN) ? env.ALLOWED_ORIGIN : '*';
  const h = new Headers(res.headers);
  h.set('Access-Control-Allow-Origin', origin);
  h.set('Access-Control-Allow-Headers', 'Content-Type, X-Api-Key');
  h.set('Access-Control-Allow-Methods', 'GET, POST, DELETE, OPTIONS');
  return new Response(res.body, { status: res.status, headers: h });
}

/**
 * NoteRoom — Durable Object
 * WebSocketでメモの変更をリアルタイム同期 + D1へ永続化
 *
 * 修正:
 *  #1 WS接続後の最初のメッセージで auth を要求（クエリパラメータ認証を廃止）
 *  #2 _expectedKey の競合状態を修正（接続ごとのローカル変数で管理）
 *  #4 送信元へのエコーバックを行わない（二重適用防止）
 *  #6 ブロードキャスト時に送信元を除外（エコーバック修正）
 *  #NEW 認証状態をタグで管理（WeakSetはDO再起動で消えるため）
 */

// タグのインデックス定義
const TAG_ROOM_ID = 0; // タグ[0]: roomId
const TAG_META = 1; // タグ[1]: JSON { expectedKey, authed }

export class NoteRoom {
  constructor(state, env) {
    this.state = state;
    this.env = env;
    this._dbReady = null;
    // _authedSockets (WeakSet) を廃止。
    // 認証状態はタグ内の authed フラグで管理し、DO再起動後も維持される。
  }

  async fetch(request) {
    const url = new URL(request.url);
    const parts = url.pathname.split('/').filter(Boolean);

    if (parts[0] === 'ws') {
      if (request.headers.get('Upgrade') !== 'websocket') {
        return new Response('WebSocket required', { status: 426 });
      }
      return this._handleWS(request);
    }

    return new Response('not found', { status: 404 });
  }

  // ---------- WebSocket ----------

  async _handleWS(request) {
    const expectedKey = request.headers.get('X-Api-Key-Expected') || null;

    const [client, server] = Object.values(new WebSocketPair());
    const parts = new URL(request.url).pathname.split('/').filter(Boolean);
    const roomId = parts[1] || '';

    // 認証状態(authed)と expectedKey を両方タグに保存する
    // DO再起動後もタグは永続するので認証状態が失われない
    this.state.acceptWebSocket(server, [
      roomId,
      JSON.stringify({ expectedKey, authed: false })
    ]);

    return new Response(null, { status: 101, webSocket: client });
  }

  // タグのメタ情報を取得
  _getMeta(ws) {
    const tags = this.state.getTags(ws);
    if (!tags || !tags[TAG_META]) return { expectedKey: null, authed: false };
    try { return JSON.parse(tags[TAG_META]); } catch { return { expectedKey: null, authed: false }; }
  }

  // 認証済みフラグをタグに書き込む（タグ自体は変更不可なので ws.serializeAttachment を使う）
  _setAuthed(ws) {
    // Cloudflare DOのタグは変更不可のため、attachmentで認証状態を管理する
    ws.serializeAttachment({ authed: true });
  }

  _isAuthed(ws) {
    try {
      const att = ws.deserializeAttachment();
      return !!(att && att.authed);
    } catch { return false; }
  }

  async webSocketMessage(ws, message) {
    try {
      const msg = JSON.parse(message);
      const tags = this.state.getTags(ws);
      const roomId = (tags && tags[TAG_ROOM_ID]) || '';
      const meta = this._getMeta(ws);

      // 未認証ソケットは auth メッセージのみ受け付ける
      if (!this._isAuthed(ws)) {
        if (msg.type === 'auth') {
          const ok = !meta.expectedKey || msg.key === meta.expectedKey;
          if (ok) {
            this._setAuthed(ws);
            ws.send(JSON.stringify({ type: 'auth_ok' }));
          } else {
            ws.send(JSON.stringify({ type: 'auth_error', reason: 'invalid key' }));
            ws.close(4001, 'unauthorized');
          }
        }
        return;
      }

      if (msg.type === 'ping') {
        ws.send(JSON.stringify({ type: 'pong' }));
        return;
      }

      if (msg.type === 'sync' && msg.action) {
        const result = await this._handleSync(msg, roomId);
        if (!result.ok) {
          ws.send(JSON.stringify({
            type: 'sync_error',
            clientMsgId: msg.clientMsgId || null,
            action: msg.action,
            error: result.error || 'sync failed'
          }));
          return;
        }
        // [fix #6] 送信元(ws)を除外してブロードキャスト（エコーバック防止）
        const json = JSON.stringify(msg);
        for (const s of this.state.getWebSockets()) {
          if (s !== ws && s.readyState === WebSocket.OPEN && this._isAuthed(s)) {
            s.send(json);
          }
        }
      }
    } catch (e) {
      console.warn('ws message error', e);
    }
  }

  async webSocketClose() { }
  async webSocketError() { }

  // ---------- DB ----------

  async _ensureDb() {
    if (!this.env.DB) return;
    if (!this._dbReady) {
      this._dbReady = (async () => {
        const stmts = [
          `CREATE TABLE IF NOT EXISTS notes (
            id TEXT PRIMARY KEY,
            room_id TEXT NOT NULL,
            title TEXT NOT NULL DEFAULT '',
            pinned INTEGER NOT NULL DEFAULT 0,
            updated_at TEXT NOT NULL,
            item_order TEXT
          )`,
          `CREATE TABLE IF NOT EXISTS groups (
            id TEXT PRIMARY KEY,
            note_id TEXT NOT NULL,
            room_id TEXT NOT NULL,
            title TEXT NOT NULL DEFAULT 'グループ',
            created_at TEXT NOT NULL
          )`,
          `CREATE TABLE IF NOT EXISTS blocks (
            id TEXT PRIMARY KEY,
            note_id TEXT NOT NULL,
            room_id TEXT NOT NULL,
            text TEXT NOT NULL DEFAULT '',
            pinned INTEGER NOT NULL DEFAULT 0,
            created_at TEXT NOT NULL,
            updated_at TEXT NOT NULL,
            group_id TEXT,
            group_order INTEGER
          )`,
          `CREATE INDEX IF NOT EXISTS idx_notes_room  ON notes(room_id, updated_at)`,
          `CREATE INDEX IF NOT EXISTS idx_groups_note ON groups(note_id)`,
          `CREATE INDEX IF NOT EXISTS idx_blocks_note ON blocks(note_id, created_at)`,
        ];
        for (const sql of stmts) {
          await this.env.DB.prepare(sql).run().catch(() => { });
        }
        // 既存DBへのカラム追加（ALTER TABLE は既存カラムがあるとエラーになるので握り潰す）
        const alters = [
          `ALTER TABLE notes  ADD COLUMN item_order  TEXT`,
          `ALTER TABLE blocks ADD COLUMN group_id    TEXT`,
          `ALTER TABLE blocks ADD COLUMN group_order INTEGER`,
        ];
        for (const sql of alters) {
          await this.env.DB.prepare(sql).run().catch(() => { });
        }
      })().catch(e => { this._dbReady = null; throw e; });
    }
    await this._dbReady;
  }

  async _handleSync(msg, roomId) {
    if (!this.env.DB) return { ok: false, error: 'DB not configured' };
    try {
      await this._ensureDb();
      const now = new Date().toISOString();
      const { action, payload } = msg;
      const noteUpdatedAt = payload.noteUpdatedAt || payload.updatedAt || now;

      // ---------- note_upsert ----------
      if (action === 'note_upsert') {
        payload.updatedAt = payload.updatedAt || now;
        const itemOrderJson = payload.itemOrder
          ? JSON.stringify(payload.itemOrder)
          : null;

        await this.env.DB.prepare(
          `INSERT INTO notes (id, room_id, title, pinned, updated_at, item_order)
           VALUES (?, ?, ?, ?, ?, ?)
           ON CONFLICT(id) DO UPDATE SET
             title      = excluded.title,
             pinned     = excluded.pinned,
             updated_at = excluded.updated_at,
             item_order = excluded.item_order
           WHERE excluded.updated_at >= notes.updated_at`
        ).bind(
          payload.id, roomId,
          payload.title || '',
          payload.pinned ? 1 : 0,
          payload.updatedAt,
          itemOrderJson
        ).run();

        // groups を全置換（差分管理より確実）
        if (payload.groups && Array.isArray(payload.groups)) {
          await this.env.DB.prepare(
            `DELETE FROM groups WHERE note_id=? AND room_id=?`
          ).bind(payload.id, roomId).run();

          for (const g of payload.groups) {
            await this.env.DB.prepare(
              `INSERT OR IGNORE INTO groups (id, note_id, room_id, title, created_at)
               VALUES (?, ?, ?, ?, ?)`
            ).bind(g.id, payload.id, roomId, g.title || 'グループ', now).run();
            // タイトル更新
            await this.env.DB.prepare(
              `UPDATE groups SET title=? WHERE id=? AND room_id=?`
            ).bind(g.title || 'グループ', g.id, roomId).run();
          }
        }

        // note_upsert に blocks が含まれる場合、groupId/groupOrder を同期
        if (payload.blocks && Array.isArray(payload.blocks)) {
          for (const b of payload.blocks) {
            await this.env.DB.prepare(
              `INSERT INTO blocks (id, note_id, room_id, text, pinned, created_at, updated_at, group_id, group_order)
               VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
               ON CONFLICT(id) DO UPDATE SET
                 group_id    = excluded.group_id,
                 group_order = excluded.group_order`
            ).bind(
              b.id,
              payload.id,
              roomId,
              b.text || '',
              b.pinned ? 1 : 0,
              b.createdAt || now,
              b.updatedAt || now,
              b.groupId || null,
              b.groupOrder ?? null
            ).run();
          }
        }
      }

      // ---------- note_delete ----------
      if (action === 'note_delete') {
        await this.env.DB.prepare(
          `DELETE FROM notes WHERE id=? AND room_id=?`
        ).bind(payload.id, roomId).run();
        await this.env.DB.prepare(
          `DELETE FROM blocks WHERE note_id=? AND room_id=?`
        ).bind(payload.id, roomId).run();
        await this.env.DB.prepare(
          `DELETE FROM groups WHERE note_id=? AND room_id=?`
        ).bind(payload.id, roomId).run();
      }

      // ---------- block_upsert ----------
      if (action === 'block_upsert') {
        payload.createdAt = payload.createdAt || now;
        payload.updatedAt = payload.updatedAt || now;
        payload.noteUpdatedAt = noteUpdatedAt;
        await this.env.DB.prepare(
          `INSERT INTO blocks (id, note_id, room_id, text, pinned, created_at, updated_at, group_id, group_order)
           VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
           ON CONFLICT(id) DO UPDATE SET
             created_at  = excluded.created_at,
             text        = excluded.text,
             pinned      = excluded.pinned,
             updated_at  = excluded.updated_at,
             group_id    = excluded.group_id,
             group_order = excluded.group_order
           WHERE excluded.updated_at >= blocks.updated_at`
        ).bind(
          payload.id,
          payload.noteId,
          roomId,
          payload.text || '',
          payload.pinned ? 1 : 0,
          payload.createdAt,
          payload.updatedAt,
          payload.groupId || null,
          payload.groupOrder ?? null
        ).run();
        await this.env.DB.prepare(
          `UPDATE notes SET updated_at=? WHERE id=? AND room_id=? AND updated_at <= ?`
        ).bind(payload.noteUpdatedAt, payload.noteId, roomId, payload.noteUpdatedAt).run();
      }

      // ---------- block_delete ----------
      if (action === 'block_delete') {
        payload.noteUpdatedAt = noteUpdatedAt;
        await this.env.DB.prepare(
          `DELETE FROM blocks WHERE id=? AND room_id=?`
        ).bind(payload.id, roomId).run();
        await this.env.DB.prepare(
          `UPDATE notes SET updated_at=? WHERE id=? AND room_id=? AND updated_at <= ?`
        ).bind(payload.noteUpdatedAt, payload.noteId, roomId, payload.noteUpdatedAt).run();
      }
      msg.serverAckAt = new Date().toISOString();
      return { ok: true };
    } catch (e) {
      console.warn('DB sync error', e);
      return { ok: false, error: e && e.message ? e.message : 'DB sync error' };
    }
  }
}
